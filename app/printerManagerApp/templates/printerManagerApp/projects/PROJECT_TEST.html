Descripción:

En esta expansión vamos a combinar las otras dos expansiones junto con una pala impresa para hacer un robot luchador de sumo. El objetivo de esta expansión es hacer que SMARS sea capaz de localizar un oponente y sacarlo fuera de un recinto delimitado mientras intenta evitar salirse del mismo recinto. Para ello haremos uso de los sensores IR y del sensor ultrasónico.


Lista de materiales:

En esta expansión no se requiere de material adicional más que los sensores montados de las otras expansiones y el archivo impreso de esta.





Paso 1:

El primer paso será desconectar todos los cables de la placa de control para poder sacarla entera para poder montar las nuevas juntas para los accesorios.

Paso 2:

Luego desmontar el sensor ultrasónico y acoplar la nueva junta para poder unirse a la pala.

Paso 3:

Finalmente volvemos a introducir la placa de electrónica cableamos los motores y la batería como estaba antes y cableamos todos los sensores como se muestra en la foto y se describe a continuación:

Segun el codigo de colores de la foto las conexiones son:

Sensor ultrasónico:
  • Blanco - +5
  • Rojo - GND
  • Negro - A5
  • Marron - A4
Sensores IR:
  • Rojo - +5
  • Naranja - GND
  • Amarillo - A3
  • Morado - +5
  • Azul - GND
  • Verde - A2
  • Morado - +5
  • Azul - GND
  • Verde - A1

Codigo Demo 5:

Este código hará que el robot siga una línea negra sin salirse, se recomienda hacer un circuito con cinta aislante negra. 

//DAMA3D
//SMARS Demo 4 ultrasonic sensor

#include <AFMotor.h> 

AF_DCMotor R_motor(4);           // defines Right motor connector
AF_DCMotor L_motor(3);           // defines Left motor connector

//IR Sensor setup:
 const int irCenterPin = A4; // signal center IR sensor connected to Arduino's pin A2
 const int irRightPin = A3;  // signal right IR sensor connected to Arduino's pin A3
 const int irLeftPin = A5;
 // signal left IR sensor connected to Arduino's pin A4
 boolean stateCenterIR = 0; // state to store irCenterPin detected
 boolean stateRightIR = 0;  // state to store irCenterPin irRightPin
 boolean stateLeftIR = 0;   // state to store irCenterPin irLeftPin

void setup() {
  Serial.begin(9600);           // sets up Serial library at 9600 bps
  pinMode(irCenterPin, INPUT);   // Sets the irCenterPin as an Input
  pinMode(irRightPin, INPUT);   // Sets the irRightPin as an Input
  pinMode(stateLeftIR, INPUT);   // Sets the stateLeftIR as an Input

//changes the following values to make the robot  drive as straight as possible   
  
  L_motor.setSpeed(200);        // sets L motor speed
  R_motor.setSpeed(200);        // sets R motor speed
  
//configuration parameters

  R_motor.run(RELEASE);         //turns L motor on
  L_motor.run(RELEASE);         //turns R motor on
}

 
//function to make the robot move forward
void goForward(){
  L_motor.run(FORWARD);      
  R_motor.run(FORWARD); 
}

//function to make the robot move Backward 
void goBackward(){
  L_motor.run(BACKWARD);      
  R_motor.run(BACKWARD); 
}

//function to make the robot turn left
void turnLeft(){
  L_motor.run(BACKWARD);      
  R_motor.run(FORWARD); 
}

//function to make the robot turn right
void turnRight(){
  L_motor.run(FORWARD);      
  R_motor.run(BACKWARD); 
}

//function to make the robot stop moving
void Stop(){
  R_motor.run(RELEASE);         
  L_motor.run(RELEASE);
}

void loop() {
if((digitalRead(irLeftPin)==false) && (digitalRead(irCenterPin)==true) && (digitalRead(irRightPin)==false)){
    
    goForward();
  }
  if((digitalRead(irLeftPin)==false) && (digitalRead(irCenterPin)==false) && (digitalRead(irRightPin)==true)){

    turnRight();
  }
  if((digitalRead(irLeftPin)==false) && (digitalRead(irCenterPin)==true) && (digitalRead(irRightPin)==true)){
   
    turnRight();
  }
  if((digitalRead(irLeftPin)==true) && (digitalRead(irCenterPin)==false) && (digitalRead(irRightPin)==false)){

    turnLeft();
  }
  if((digitalRead(irLeftPin)==true) && (digitalRead(irCenterPin)==true) && (digitalRead(irRightPin)==false)){

    turnLeft();
  }
  if((digitalRead(irLeftPin)==true) && (digitalRead(irCenterPin)==true) && (digitalRead(irRightPin)==true)){

     goForward();
  } 
}
